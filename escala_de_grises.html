import cv2

# Cargar la imagen en formato BGR
img = cv2.imread('equipo.jpg')  # Devuelve una matriz NumPy que representa la imagen en formato BGR (Azul, Verde, Rojo)

# Esperar hasta que el usuario presione una tecla

# Si pusieras un n√∫mero, por ejemplo 1000, esperar√≠a 1000 milisegundos (1 segundo) antes de cerrar las ventanas autom√°ticamente

# Cerrar todas las ventanas
